"""
LLM client wrapping OpenRouter API via OpenAI-compatible interface.
"""
import time
from openai import OpenAI
from loguru import logger
from config.settings import OPENROUTER_API_KEY, OPENROUTER_BASE_URL, PRIMARY_MODEL

client = OpenAI(
    base_url=OPENROUTER_BASE_URL,
    api_key=OPENROUTER_API_KEY,
)


def call_llm(
    prompt: str,
    system_prompt: str = "You are a senior geotechnical engineer in Hong Kong.",
    model: str = PRIMARY_MODEL,
    temperature: float = 0.2,
    max_tokens: int = 4000,
    retries: int = 3,
) -> str:
    """
    Call OpenRouter LLM and return the response text.
    """
    for attempt in range(retries):
        try:
            response = client.chat.completions.create(
                model=model,
                messages=[
                    {"role": "system", "content": system_prompt},
                    {"role": "user", "content": prompt},
                ],
                temperature=temperature,
                max_tokens=max_tokens,
                extra_headers={
                    "HTTP-Referer": "https://geotech-agent.local",
                    "X-Title": "Geotech AI Agent",
                },
            )
            return response.choices[0].message.content
        except Exception as e:
            logger.warning(f"LLM call attempt {attempt + 1} failed: {e}")
            if attempt < retries - 1:
                time.sleep(2 ** attempt)
            else:
                raise


def call_llm_with_context(
    question: str,
    context_chunks: list[dict],
    model: str = PRIMARY_MODEL,
) -> str:
    """
    RAG-style call: provide retrieved context chunks and a question.
    """
    context_text = "\n\n---\n\n".join(
        [
            f"[Source: {c['metadata'].get('document_name', 'Unknown')} | "
            f"Section: {c['metadata'].get('section_title', 'N/A')} | "
            f"Page: {c['metadata'].get('page_number', 'N/A')}]\n{c['text']}"
            for c in context_chunks
        ]
    )

    system_prompt = """You are a senior geotechnical engineer working in Hong Kong.
You provide answers based ONLY on the provided reference documents.
You cite specific clause numbers, section titles, and page numbers.
If the provided context does not contain enough information to answer,
say so explicitly. Never invent clause numbers or code requirements.

Applicable codes framework: HK Code of Practice, Geoguides, BS codes, Eurocodes.
"""

    user_prompt = f"""Based on the following reference documents, answer the question.

REFERENCE DOCUMENTS:
{context_text}

QUESTION: {question}

Provide your answer with specific citations to the source documents above.
Format citations as [Source: document name, Section/Clause: X.X.X].
"""

    return call_llm(user_prompt, system_prompt=system_prompt, model=model)

"""
Local embedding engine using sentence-transformers.
Runs entirely on your machine — no API costs, no data leakage.
"""
from sentence_transformers import SentenceTransformer
from loguru import logger
from config.settings import EMBEDDING_MODEL, EMBEDDING_DIMENSION

_model = None


def get_model() -> SentenceTransformer:
    global _model
    if _model is None:
        logger.info(f"Loading embedding model: {EMBEDDING_MODEL}")
        _model = SentenceTransformer(EMBEDDING_MODEL)
        logger.info("Embedding model loaded successfully")
    return _model


def embed_text(text: str) -> list[float]:
    """Embed a single text string."""
    model = get_model()
    # bge models benefit from a query prefix for retrieval
    embedding = model.encode(text, normalize_embeddings=True)
    return embedding.tolist()


def embed_texts(texts: list[str], batch_size: int = 32) -> list[list[float]]:
    """Embed a batch of texts."""
    model = get_model()
    embeddings = model.encode(
        texts,
        batch_size=batch_size,
        show_progress_bar=True,
        normalize_embeddings=True,
    )
    return embeddings.tolist()


def embed_query(query: str) -> list[float]:
    """
    Embed a search query.
    For BGE models, prepend instruction for better retrieval.
    """
    model = get_model()
    instruction = "Represent this sentence for searching relevant passages: "
    embedding = model.encode(
        instruction + query,
        normalize_embeddings=True,
    )
    return embedding.tolist()

"""
PDF document processor for geotechnical codes, manuals, and reports.
Extracts structured text, splits into chunks, preserves metadata.
"""
import re
import fitz  # pymupdf
import pymupdf4llm
from pathlib import Path
from dataclasses import dataclass, field
from loguru import logger


@dataclass
class DocumentChunk:
    """A single chunk of document text with metadata."""
    text: str
    metadata: dict = field(default_factory=dict)


@dataclass
class ProcessedDocument:
    """A fully processed document ready for embedding."""
    document_id: str
    document_name: str
    document_type: str  # "code", "manual", "report", "drawing", "guideline"
    chunks: list[DocumentChunk] = field(default_factory=list)


def extract_text_from_pdf(pdf_path: str) -> str:
    """Extract markdown-formatted text from PDF using pymupdf4llm."""
    logger.info(f"Extracting text from: {pdf_path}")
    try:
        md_text = pymupdf4llm.to_markdown(pdf_path, page_chunks=False)
        return md_text
    except Exception as e:
        logger.warning(f"pymupdf4llm failed, falling back to basic extraction: {e}")
        return _basic_extract(pdf_path)


def _basic_extract(pdf_path: str) -> str:
    """Fallback: basic text extraction with pymupdf."""
    doc = fitz.open(pdf_path)
    text_parts = []
    for page_num, page in enumerate(doc, 1):
        text = page.get_text("text")
        text_parts.append(f"\n[Page {page_num}]\n{text}")
    doc.close()
    return "\n".join(text_parts)


def extract_pages_with_numbers(pdf_path: str) -> list[dict]:
    """Extract text page by page with page numbers."""
    doc = fitz.open(pdf_path)
    pages = []
    for page_num, page in enumerate(doc, 1):
        text = page.get_text("text")
        if text.strip():
            pages.append({
                "page_number": page_num,
                "text": text.strip(),
            })
    doc.close()
    return pages


def chunk_by_sections(
    full_text: str,
    document_id: str,
    document_name: str,
    document_type: str,
    max_chunk_tokens: int = 800,
    overlap_tokens: int = 100,
) -> ProcessedDocument:
    """
    Split document text into chunks by section headings.
    
    Strategy:
    1. First try to split by markdown headings (##, ###)
    2. If sections are too large, sub-split by paragraphs
    3. Preserve section title in metadata
    """
    # Split by markdown headings
    heading_pattern = r'^(#{1,4})\s+(.+)$'
    lines = full_text.split('\n')
    
    sections = []
    current_section_title = "Preamble"
    current_section_level = 0
    current_text_lines = []
    
    for line in lines:
        heading_match = re.match(heading_pattern, line, re.MULTILINE)
        if heading_match:
            # Save previous section
            if current_text_lines:
                section_text = '\n'.join(current_text_lines).strip()
                if section_text:
                    sections.append({
                        "title": current_section_title,
                        "level": current_section_level,
                        "text": section_text,
                    })
            # Start new section
            current_section_level = len(heading_match.group(1))
            current_section_title = heading_match.group(2).strip()
            current_text_lines = []
        else:
            current_text_lines.append(line)
    
    # Don't forget the last section
    if current_text_lines:
        section_text = '\n'.join(current_text_lines).strip()
        if section_text:
            sections.append({
                "title": current_section_title,
                "level": current_section_level,
                "text": section_text,
            })
    
    # If no headings found, split by page markers or fixed size
    if len(sections) <= 1:
        sections = _split_by_pages_or_size(full_text, max_chunk_tokens)
    
    # Now create chunks, sub-splitting large sections
    doc = ProcessedDocument(
        document_id=document_id,
        document_name=document_name,
        document_type=document_type,
    )
    
    for section in sections:
        section_text = section["text"]
        estimated_tokens = len(section_text.split()) * 1.3  # rough token estimate
        
        if estimated_tokens <= max_chunk_tokens:
            # Section fits in one chunk
            doc.chunks.append(DocumentChunk(
                text=section_text,
                metadata={
                    "document_id": document_id,
                    "document_name": document_name,
                    "document_type": document_type,
                    "section_title": section.get("title", ""),
                    "section_level": section.get("level", 0),
                },
            ))
        else:
            # Sub-split large sections by paragraphs
            sub_chunks = _split_large_section(
                section_text,
                section.get("title", ""),
                max_chunk_tokens,
                overlap_tokens,
            )
            for i, sub_text in enumerate(sub_chunks):
                doc.chunks.append(DocumentChunk(
                    text=sub_text,
                    metadata={
                        "document_id": document_id,
                        "document_name": document_name,
                        "document_type": document_type,
                        "section_title": section.get("title", ""),
                        "section_level": section.get("level", 0),
                        "sub_chunk_index": i,
                    },
                ))
    
    logger.info(
        f"Processed '{document_name}': {len(doc.chunks)} chunks from {len(sections)} sections"
    )
    return doc


def _split_large_section(
    text: str, title: str, max_tokens: int, overlap_tokens: int
) -> list[str]:
    """Split a large section into overlapping chunks by paragraphs."""
    paragraphs = text.split('\n\n')
    chunks = []
    current_chunk_parts = []
    current_token_count = 0
    
    for para in paragraphs:
        para_tokens = len(para.split()) * 1.3
        
        if current_token_count + para_tokens > max_tokens and current_chunk_parts:
            chunks.append('\n\n'.join(current_chunk_parts))
            # Keep last paragraph as overlap
            overlap_parts = current_chunk_parts[-1:] if overlap_tokens > 0 else []
            current_chunk_parts = overlap_parts
            current_token_count = sum(len(p.split()) * 1.3 for p in current_chunk_parts)
        
        current_chunk_parts.append(para)
        current_token_count += para_tokens
    
    if current_chunk_parts:
        chunks.append('\n\n'.join(current_chunk_parts))
    
    return chunks


def _split_by_pages_or_size(text: str, max_tokens: int) -> list[dict]:
    """Fallback splitting when no headings are found."""
    # Try page markers first
    page_pattern = r'\[Page (\d+)\]'
    page_splits = re.split(page_pattern, text)
    
    if len(page_splits) > 1:
        sections = []
        for i in range(1, len(page_splits), 2):
            page_num = page_splits[i]
            page_text = page_splits[i + 1] if i + 1 < len(page_splits) else ""
            sections.append({
                "title": f"Page {page_num}",
                "level": 0,
                "text": page_text.strip(),
            })
        return sections
    
    # Last resort: fixed-size chunks
    words = text.split()
    chunk_word_count = int(max_tokens / 1.3)
    sections = []
    for i in range(0, len(words), chunk_word_count):
        chunk_text = ' '.join(words[i:i + chunk_word_count])
        sections.append({
            "title": f"Chunk {i // chunk_word_count + 1}",
            "level": 0,
            "text": chunk_text,
        })
    return sections

"""
Qdrant vector database interface for storing and searching document chunks.
"""
import uuid
from qdrant_client import QdrantClient
from qdrant_client.http.models import (
    Distance,
    VectorParams,
    PointStruct,
    Filter,
    FieldCondition,
    MatchValue,
    models,
)
from loguru import logger
from config.settings import (
    QDRANT_HOST,
    QDRANT_PORT,
    COLLECTION_NAME,
    EMBEDDING_DIMENSION,
)
from src.utils.embeddings import embed_texts, embed_query


class GeoVectorStore:
    """Vector store for geotechnical knowledge base."""

    def __init__(
        self,
        host: str = QDRANT_HOST,
        port: int = QDRANT_PORT,
        collection_name: str = COLLECTION_NAME,
        use_local: bool = False,
    ):
        """
        Initialize Qdrant client.
        
        Args:
            use_local: If True, use file-based storage instead of Docker.
                       Useful if Docker is not available.
        """
        if use_local:
            self.client = QdrantClient(path="./qdrant_local_storage")
            logger.info("Using local file-based Qdrant storage")
        else:
            self.client = QdrantClient(host=host, port=port)
            logger.info(f"Connected to Qdrant at {host}:{port}")

        self.collection_name = collection_name
        self._ensure_collection()

    def _ensure_collection(self):
        """Create collection if it doesn't exist."""
        collections = self.client.get_collections().collections
        exists = any(c.name == self.collection_name for c in collections)

        if not exists:
            self.client.create_collection(
                collection_name=self.collection_name,
                vectors_config=VectorParams(
                    size=EMBEDDING_DIMENSION,
                    distance=Distance.COSINE,
                ),
            )
            # Create payload indexes for filtering
            self.client.create_payload_index(
                collection_name=self.collection_name,
                field_name="document_type",
                field_schema=models.PayloadSchemaType.KEYWORD,
            )
            self.client.create_payload_index(
                collection_name=self.collection_name,
                field_name="document_id",
                field_schema=models.PayloadSchemaType.KEYWORD,
            )
            logger.info(f"Created collection: {self.collection_name}")
        else:
            logger.info(f"Collection already exists: {self.collection_name}")

    def add_document(self, processed_doc) -> int:
        """
        Add a processed document's chunks to the vector store.
        
        Args:
            processed_doc: ProcessedDocument instance
            
        Returns:
            Number of chunks added
        """
        if not processed_doc.chunks:
            logger.warning(f"No chunks to add for {processed_doc.document_name}")
            return 0

        texts = [chunk.text for chunk in processed_doc.chunks]
        logger.info(f"Generating embeddings for {len(texts)} chunks...")
        embeddings = embed_texts(texts)

        points = []
        for chunk, embedding in zip(processed_doc.chunks, embeddings):
            point_id = str(uuid.uuid4())
            payload = {
                "text": chunk.text,
                **chunk.metadata,
            }
            points.append(
                PointStruct(id=point_id, vector=embedding, payload=payload)
            )

        # Upsert in batches of 100
        batch_size = 100
        for i in range(0, len(points), batch_size):
            batch = points[i : i + batch_size]
            self.client.upsert(
                collection_name=self.collection_name,
                points=batch,
            )

        logger.info(
            f"Added {len(points)} chunks from '{processed_doc.document_name}'"
        )
        return len(points)

    def search(
        self,
        query: str,
        top_k: int = 10,
        document_type: str | None = None,
        document_id: str | None = None,
        score_threshold: float = 0.3,
    ) -> list[dict]:
        """
        Semantic search for relevant document chunks.
        
        Args:
            query: Search query text
            top_k: Number of results to return
            document_type: Filter by type ("code", "manual", "report", etc.)
            document_id: Filter by specific document
            score_threshold: Minimum similarity score
            
        Returns:
            List of dicts with 'text', 'metadata', and 'score'
        """
        query_embedding = embed_query(query)

        # Build filters
        filter_conditions = []
        if document_type:
            filter_conditions.append(
                FieldCondition(
                    key="document_type", match=MatchValue(value=document_type)
                )
            )
        if document_id:
            filter_conditions.append(
                FieldCondition(
                    key="document_id", match=MatchValue(value=document_id)
                )
            )

        search_filter = (
            Filter(must=filter_conditions) if filter_conditions else None
        )

        results = self.client.search(
            collection_name=self.collection_name,
            query_vector=query_embedding,
            limit=top_k,
            query_filter=search_filter,
            score_threshold=score_threshold,
        )

        return [
            {
                "text": hit.payload.get("text", ""),
                "metadata": {
                    k: v for k, v in hit.payload.items() if k != "text"
                },
                "score": hit.score,
            }
            for hit in results
        ]

    def list_documents(self) -> list[dict]:
        """List all documents in the collection with chunk counts."""
        # Scroll through all points and aggregate by document_id
        all_points, _ = self.client.scroll(
            collection_name=self.collection_name,
            limit=10000,
            with_payload=["document_id", "document_name", "document_type"],
        )

        docs = {}
        for point in all_points:
            doc_id = point.payload.get("document_id", "unknown")
            if doc_id not in docs:
                docs[doc_id] = {
                    "document_id": doc_id,
                    "document_name": point.payload.get("document_name", ""),
                    "document_type": point.payload.get("document_type", ""),
                    "chunk_count": 0,
                }
            docs[doc_id]["chunk_count"] += 1

        return list(docs.values())

    def delete_document(self, document_id: str):
        """Remove all chunks for a specific document."""
        self.client.delete(
            collection_name=self.collection_name,
            points_selector=models.FilterSelector(
                filter=Filter(
                    must=[
                        FieldCondition(
                            key="document_id",
                            match=MatchValue(value=document_id),
                        )
                    ]
                )
            ),
        )
        logger.info(f"Deleted all chunks for document: {document_id}")

class QAAgent:
    """
    Question-answering agent that searches the vector database
    and provides answers grounded in your codes and manuals.
    """

    def __init__(self, use_local_db: bool = False):
        self.store = GeoVectorStore(use_local=use_local_db)
        self.conversation_history: list[dict] = []

    def ask(
        self,
        question: str,
        document_type: str | None = None,
        document_id: str | None = None,
        top_k: int = 8,
        model: str = PRIMARY_MODEL,
    ) -> dict:
        """
        Ask a question and get an answer grounded in your documents.
        
        Args:
            question: The user's question
            document_type: Optionally filter by document type
            document_id: Optionally filter by specific document
            top_k: Number of chunks to retrieve
            model: Which LLM model to use
            
        Returns:
            dict with 'answer', 'sources', and 'query'
        """
        # Step 1: Retrieve relevant chunks
        logger.info(f"Searching for: {question}")
        chunks = self.store.search(
            query=question,
            top_k=top_k,
            document_type=document_type,
            document_id=document_id,
        )

        if not chunks:
            return {
                "answer": "No relevant information found in the knowledge base. "
                          "Please check that the relevant documents have been ingested.",
                "sources": [],
                "query": question,
            }

        # Step 2: Generate answer using LLM with context
        answer = call_llm_with_context(
            question=question,
            context_chunks=chunks,
            model=model,
        )

        # Step 3: Format sources for citation
        sources = []
        seen = set()
        for chunk in chunks:
            source_key = (
                chunk["metadata"].get("document_name", ""),
                chunk["metadata"].get("section_title", ""),
            )
            if source_key not in seen:
                seen.add(source_key)
                sources.append({
                    "document": chunk["metadata"].get("document_name", "Unknown"),
                    "section": chunk["metadata"].get("section_title", "N/A"),
                    "page": chunk["metadata"].get("page_number", "N/A"),
                    "relevance_score": round(chunk["score"], 3),
                })

        # Store in conversation history
        self.conversation_history.append({
            "question": question,
            "answer": answer,
            "sources": sources,
        })

        return {
            "answer": answer,
            "sources": sources,
            "query": question,
        }

"""
Bearing capacity calculations for shallow foundations.
Deterministic functions — NO LLM involvement in computation.

References:
- Code of Practice for Foundations 2017 (HK)
- Geoguide 1
- BS EN 1997-1 (Eurocode 7)
"""
import math
from dataclasses import dataclass


@dataclass
class SoilParameters:
    """Characteristic soil parameters for bearing capacity."""
    cohesion_kpa: float  # c' (effective) or cu (undrained)
    friction_angle_deg: float  # φ' in degrees
    unit_weight_kn_m3: float  # γ (bulk unit weight)
    is_undrained: bool = False  # True for total stress analysis


@dataclass
class FoundationGeometry:
    """Foundation dimensions."""
    width_m: float  # B
    length_m: float  # L (use large number for strip footing)
    depth_m: float  # Df (embedment depth)


@dataclass
class GroundwaterCondition:
    """Groundwater information."""
    gwl_depth_m: float  # Depth to GWL from ground surface
    water_unit_weight_kn_m3: float = 9.81


@dataclass
class BearingCapacityResult:
    """Result of bearing capacity calculation."""
    q_ult_kpa: float  # Ultimate bearing capacity
    q_allowable_kpa: float  # Allowable bearing capacity (with FoS)
    factor_of_safety: float
    method: str
    bearing_capacity_factors: dict  # Nc, Nq, Nγ
    shape_factors: dict  # sc, sq, sγ
    depth_factors: dict  # dc, dq, dγ
    notes: list[str]


def terzaghi_bearing_capacity(
    soil: SoilParameters,
    foundation: FoundationGeometry,
    gwl: GroundwaterCondition | None = None,
    fos: float = 3.0,
) -> BearingCapacityResult:
    """
    Terzaghi's bearing capacity equation for shallow foundations.
    
    q_ult = c·Nc·sc + q·Nq + 0.5·γ·B·Nγ·sγ
    
    For strip footing: sc = 1.0, sγ = 1.0
    For square footing: sc = 1.3, sγ = 0.8
    For circular footing: sc = 1.3, sγ = 0.6
    """
    phi_rad = math.radians(soil.friction_angle_deg)
    
    # Bearing capacity factors (Terzaghi)
    if soil.friction_angle_deg == 0:
        Nc = 5.14
        Nq = 1.0
        Ngamma = 0.0
    else:
        Nq = math.exp(math.pi * math.tan(phi_rad)) * (
            math.tan(math.radians(45 + soil.friction_angle_deg / 2)) ** 2
        )
        Nc = (Nq - 1) / math.tan(phi_rad)
        # Terzaghi's Nγ (using Kumbhojkar's approximation)
        Ngamma = 2 * (Nq + 1) * math.tan(phi_rad)
    
    # Shape factors (Terzaghi)
    B = foundation.width_m
    L = foundation.length_m
    
    if L / B > 10:  # Strip footing
        sc, sgamma = 1.0, 1.0
    elif abs(L - B) < 0.01:  # Square
        sc, sgamma = 1.3, 0.8
    else:  # Rectangular
        sc = 1.0 + 0.3 * (B / L)
        sgamma = 1.0 - 0.2 * (B / L)
    
    sq = 1.0  # Terzaghi doesn't have sq
    
    # Overburden pressure at foundation level
    gamma_eff = soil.unit_weight_kn_m3
    gamma_below = soil.unit_weight_kn_m3
    
    notes = []
    
    # Groundwater correction
    if gwl is not None:
        if gwl.gwl_depth_m <= foundation.depth_m:
            # GWL above or at foundation level
            gamma_eff_above = (
                soil.unit_weight_kn_m3 - gwl.water_unit_weight_kn_m3
            )
            gamma_below = soil.unit_weight_kn_m3 - gwl.water_unit_weight_kn_m3
            q = gamma_eff_above * foundation.depth_m
            notes.append(
                f"GWL at {gwl.gwl_depth_m}m (above foundation level). "
                f"Using submerged unit weight."
            )
        elif gwl.gwl_depth_m < foundation.depth_m + B:
            # GWL within influence zone below foundation
            q = soil.unit_weight_kn_m3 * foundation.depth_m
            d_below = gwl.gwl_depth_m - foundation.depth_m
            gamma_below = soil.unit_weight_kn_m3 - (
                gwl.water_unit_weight_kn_m3 * (1 - d_below / B)
            )
            notes.append(
                f"GWL at {gwl.gwl_depth_m}m (within B below foundation). "
                f"Interpolated unit weight used."
            )
        else:
            q = soil.unit_weight_kn_m3 * foundation.depth_m
            notes.append(
                f"GWL at {gwl.gwl_depth_m}m (below influence zone). "
                f"No groundwater correction needed."
            )
    else:
        q = soil.unit_weight_kn_m3 * foundation.depth_m
        notes.append("No groundwater information provided.")
    
    # Calculate ultimate bearing capacity
    q_ult = (
        soil.cohesion_kpa * Nc * sc
        + q * Nq * sq
        + 0.5 * gamma_below * B * Ngamma * sgamma
    )
    
    q_allowable = q_ult / fos
    
    notes.append(f"Factor of Safety applied: {fos}")
    
    return BearingCapacityResult(
        q_ult_kpa=round(q_ult, 1),
        q_allowable_kpa=round(q_allowable, 1),
        factor_of_safety=fos,
        method="Terzaghi",
        bearing_capacity_factors={"Nc": round(Nc, 2), "Nq": round(Nq, 2), "Nγ": round(Ngamma, 2)},
        shape_factors={"sc": round(sc, 2), "sq": round(sq, 2), "sγ": round(sgamma, 2)},
        depth_factors={"dc": 1.0, "dq": 1.0, "dγ": 1.0},  # Terzaghi doesn't use depth factors
        notes=notes,
    )


def hansen_bearing_capacity(
    soil: SoilParameters,
    foundation: FoundationGeometry,
    gwl: GroundwaterCondition | None = None,
    fos: float = 3.0,
) -> BearingCapacityResult:
    """
    Hansen's general bearing capacity equation.
    
    q_ult = c·Nc·sc·dc·ic·bc·gc 
          + q·Nq·sq·dq·iq·bq·gq 
          + 0.5·γ·B'·Nγ·sγ·dγ·iγ·bγ·gγ
    
    (Inclination, base tilt, and ground slope factors set to 1.0
     unless specified. Can be extended later.)
    """
    phi_rad = math.radians(soil.friction_angle_deg)
    B = foundation.width_m
    L = foundation.length_m
    Df = foundation.depth_m
    
    # Bearing capacity factors (Hansen/Vesic)
    if soil.friction_angle_deg == 0:
        Nc = 5.14
        Nq = 1.0
        Ngamma = 0.0
    else:
        Nq = math.exp(math.pi * math.tan(phi_rad)) * (
            math.tan(math.radians(45 + soil.friction_angle_deg / 2)) ** 2
        )
        Nc = (Nq - 1) / math.tan(phi_rad)
        Ngamma = 1.5 * (Nq - 1) * math.tan(phi_rad)  # Hansen
    
    # Shape factors (Hansen)
    sc = 1.0 + (Nq / Nc) * (B / L) if soil.friction_angle_deg > 0 else 1.0 + 0.2 * (B / L)
    sq = 1.0 + (B / L) * math.tan(phi_rad)
    sgamma = 1.0 - 0.4 * (B / L)
    sgamma = max(sgamma, 0.6)  # Minimum value
    
    # Depth factors (Hansen)
    k = Df / B if Df / B <= 1 else math.atan(Df / B)
    dc = 1.0 + 0.4 * k
    dq = 1.0 + 2 * math.tan(phi_rad) * (1 - math.sin(phi_rad)) ** 2 * k
    dgamma = 1.0
    
    # Overburden and groundwater (same logic as Terzaghi)
    gamma_below = soil.unit_weight_kn_m3
    notes = []
    
    if gwl is not None and gwl.gwl_depth_m <= Df:
        gamma_eff_above = soil.unit_weight_kn_m3 - gwl.water_unit_weight_kn_m3
        q = gamma_eff_above * Df
        gamma_below = soil.unit_weight_kn_m3 - gwl.water_unit_weight_kn_m3
        notes.append(f"GWL at {gwl.gwl_depth_m}m — submerged unit weight used.")
    else:
        q = soil.unit_weight_kn_m3 * Df
        if gwl is not None:
            notes.append(f"GWL at {gwl.gwl_depth_m}m — below foundation level.")
    
    q_ult = (
        soil.cohesion_kpa * Nc * sc * dc
        + q * Nq * sq * dq
        + 0.5 * gamma_below * B * Ngamma * sgamma * dgamma
    )
    
    q_allowable = q_ult / fos
    
    return BearingCapacityResult(
        q_ult_kpa=round(q_ult, 1),
        q_allowable_kpa=round(q_allowable, 1),
        factor_of_safety=fos,
        method="Hansen",
        bearing_capacity_factors={"Nc": round(Nc, 2), "Nq": round(Nq, 2), "Nγ": round(Ngamma, 2)},
        shape_factors={"sc": round(sc, 3), "sq": round(sq, 3), "sγ": round(sgamma, 3)},
        depth_factors={"dc": round(dc, 3), "dq": round(dq, 3), "dγ": round(dgamma, 3)},
        notes=notes,
    )

@dataclass
class EC7PartialFactors:
    """Eurocode 7 partial factors for Design Approach 1."""
    gamma_c: float = 1.25  # Partial factor on c'
    gamma_phi: float = 1.25  # Partial factor on tan(φ')
    gamma_cu: float = 1.4  # Partial factor on cu
    gamma_G: float = 1.35  # Partial factor on permanent actions
    gamma_Q: float = 1.5  # Partial factor on variable actions
    gamma_Rv: float = 1.0  # Resistance factor
    combination: str = "C2"  # "C1" or "C2"


def ec7_bearing_capacity_da1(
    soil: SoilParameters,
    foundation: FoundationGeometry,
    gwl: GroundwaterCondition | None = None,
    characteristic_permanent_load_kn: float = 0,
    characteristic_variable_load_kn: float = 0,
) -> dict:
    """
    EC7 Design Approach 1 bearing capacity check.
    
    Combination 1: A1 + M1 + R1 (factor actions, don't factor soil)
    Combination 2: A2 + M2 + R1 (don't factor actions much, factor soil)
    
    Both must be satisfied.
    """
    results = {}
    
    # Combination 1: A1 + M1 + R1
    c1_factors = EC7PartialFactors(
        gamma_c=1.0, gamma_phi=1.0, gamma_cu=1.0,
        gamma_G=1.35, gamma_Q=1.5, gamma_Rv=1.0,
        combination="C1",
    )
    
    # Combination 2: A2 + M2 + R1  
    c2_factors = EC7PartialFactors(
        gamma_c=1.25, gamma_phi=1.25, gamma_cu=1.4,
        gamma_G=1.0, gamma_Q=1.3, gamma_Rv=1.0,
        combination="C2",
    )
    
    for name, factors in [("Combination_1", c1_factors), ("Combination_2", c2_factors)]:
        # Apply partial factors to soil parameters
        design_c = soil.cohesion_kpa / factors.gamma_c
        design_phi = math.degrees(
            math.atan(math.tan(math.radians(soil.friction_angle_deg)) / factors.gamma_phi)
        )
        
        design_soil = SoilParameters(
            cohesion_kpa=design_c,
            friction_angle_deg=design_phi,
            unit_weight_kn_m3=soil.unit_weight_kn_m3,
        )
        
        # Calculate design resistance using Hansen
        bc_result = hansen_bearing_capacity(
            soil=design_soil,
            foundation=foundation,
            gwl=gwl,
            fos=1.0,  # No additional FoS — partial factors handle safety
        )
        
        # Design bearing resistance
        R_d = bc_result.q_ult_kpa * foundation.width_m * foundation.length_m / factors.gamma_Rv
        
        # Design action (load)
        V_d = (
            characteristic_permanent_load_kn * factors.gamma_G
            + characteristic_variable_load_kn * factors.gamma_Q
        )
        
        # Design bearing pressure
        q_d = V_d / (foundation.width_m * foundation.length_m) if V_d > 0 else 0
        
        utilisation = V_d / R_d if R_d > 0 else float('inf')
        
        results[name] = {
            "partial_factors": factors,
            "design_soil_c": round(design_c, 1),
            "design_soil_phi": round(design_phi, 1),
            "q_ult_design_kpa": bc_result.q_ult_kpa,
            "R_d_kn": round(R_d, 1),
            "V_d_kn": round(V_d, 1),
            "q_design_kpa": round(q_d, 1),
            "utilisation_ratio": round(utilisation, 3),
            "pass": utilisation <= 1.0,
        }
    
    overall_pass = all(r["pass"] for r in results.values())
    results["overall_pass"] = overall_pass
    results["governing"] = "Combination_1" if (
        results["Combination_1"]["utilisation_ratio"]
        > results["Combination_2"]["utilisation_ratio"]
    ) else "Combination_2"
    
    return results

"""
Slope stability calculations.
Basic methods for preliminary assessment.

References:
- Geotechnical Manual for Slopes (GEO)
- Geoguide 1
"""
import math
from dataclasses import dataclass


@dataclass
class SlopeGeometry:
    """Slope geometry parameters."""
    height_m: float
    angle_deg: float  # Slope angle from horizontal
    

@dataclass
class SlopeStabilityResult:
    """Result of slope stability calculation."""
    factor_of_safety: float
    method: str
    critical_slip_surface: str
    pass_fail: bool
    required_fos: float
    notes: list[str]


def infinite_slope_drained(
    slope_angle_deg: float,
    friction_angle_deg: float,
    cohesion_kpa: float,
    unit_weight_kn_m3: float,
    depth_to_slip_m: float,
    gwl_above_slip_m: float = 0.0,
    gamma_w: float = 9.81,
    required_fos: float = 1.4,
) -> SlopeStabilityResult:
    """
    Infinite slope analysis (drained conditions).
    
    FoS = (c' + (γ·z·cos²β - γw·hw·cos²β)·tanφ') / (γ·z·sinβ·cosβ)
    
    Common for shallow translational slides in CDG/HDG (Hong Kong).
    """
    beta = math.radians(slope_angle_deg)
    phi = math.radians(friction_angle_deg)
    z = depth_to_slip_m
    hw = gwl_above_slip_m
    gamma = unit_weight_kn_m3
    c = cohesion_kpa
    
    # Normal stress on slip surface
    sigma_n = gamma * z * math.cos(beta) ** 2
    
    # Pore pressure contribution
    u = gamma_w * hw * math.cos(beta) ** 2
    
    # Shear strength (Mohr-Coulomb)
    tau_f = c + (sigma_n - u) * math.tan(phi)
    
    # Driving shear stress
    tau_d = gamma * z * math.sin(beta) * math.cos(beta)
    
    if tau_d <= 0:
        fos = float('inf')
    else:
        fos = tau_f / tau_d
    
    notes = []
    if hw > 0:
        notes.append(f"Groundwater: {hw}m above slip surface")
        ru = u / (gamma * z * math.cos(beta) ** 2)
        notes.append(f"Pore pressure ratio ru ≈ {ru:.2f}")
    
    pass_fail = fos >= required_fos
    
    return SlopeStabilityResult(
        factor_of_safety=round(fos, 3),
        method="Infinite Slope (Drained)",
        critical_slip_surface=f"Planar at {z}m depth",
        pass_fail=pass_fail,
        required_fos=required_fos,
        notes=notes,
    )


def taylor_stability_number(
    slope_angle_deg: float,
    height_m: float,
    cohesion_kpa: float,
    unit_weight_kn_m3: float,
    friction_angle_deg: float = 0.0,
    required_fos: float = 1.4,
) -> SlopeStabilityResult:
    """
    Taylor's stability chart method (φ=0 or φ>0 approximation).
    
    For undrained (φ=0) analysis:
    FoS = Ns · cu / (γ · H)
    
    where Ns is Taylor's stability number (from chart).
    
    This is an approximation — for detailed analysis, use 
    Bishop's method or numerical software.
    """
    notes = []
    
    if friction_angle_deg == 0:
        # φ = 0 analysis (undrained)
        beta = slope_angle_deg
        
        # Approximate Ns from Taylor's chart
        # These are approximate values for toe circles
        if beta >= 53:
            Ns = 5.52  # Vertical cut
        elif beta >= 45:
            Ns = 5.5 + (53 - beta) * (6.0 - 5.5) / (53 - 45)
        elif beta >= 30:
            Ns = 6.0 + (45 - beta) * (7.5 - 6.0) / (45 - 30)
        elif beta >= 15:
            Ns = 7.5 + (30 - beta) * (11.0 - 7.5) / (30 - 15)
        else:
            Ns = 11.0
            notes.append("Very gentle slope — stability number approximate.")
        
        fos = Ns * cohesion_kpa / (unit_weight_kn_m3 * height_m)
        notes.append(f"Taylor's stability number Ns ≈ {Ns:.2f}")
        notes.append("Undrained (φ=0) analysis — short-term condition")
    else:
        notes.append("For φ>0, Taylor's method requires chart interpolation.")
        notes.append("This is an approximate calculation. Use Bishop's method for detailed analysis.")
        # Very rough approximation
        phi_rad = math.radians(friction_angle_deg)
        c_component = cohesion_kpa / (unit_weight_kn_m3 * height_m)
        phi_component = math.tan(phi_rad) / math.tan(math.radians(slope_angle_deg))
        fos = c_component * 5.5 + phi_component  # Rough approximation
    
    pass_fail = fos >= required_fos
    
    return SlopeStabilityResult(
        factor_of_safety=round(fos, 3),
        method="Taylor's Stability Number",
        critical_slip_surface="Circular (toe circle assumed)",
        pass_fail=pass_fail,
        required_fos=required_fos,
        notes=notes,
    )

"""
Skill executor: runs deterministic calculations and formats results.
The LLM NEVER computes — it only describes results.
"""
import importlib
from loguru import logger
from src.skills.catalog import SkillCatalog
from src.skills.skill_models import Skill
from src.calculations.bearing_capacity import (
    SoilParameters,

    ec7_bearing_capacity_da1,
)
from src.calculations.slope_stability import (
    infinite_slope_drained,
    taylor_stability_number,
)


class SkillExecutor:
    """Execute skills using deterministic calculation functions."""
    
    def __init__(self, catalog: SkillCatalog):
        self.catalog = catalog
        # Map skill_ids to their actual Python functions
        self._function_map = {
            "shallow_bearing_capacity": self._execute_bearing_capacity,
            "slope_stability_infinite": self._execute_slope_stability,
            "earth_pressure_rankine": self._execute_earth_pressure,
        }
    
    def execute(self, skill_id: str, inputs: dict) -> dict:
        """
        Execute a skill with the given inputs.
        
        Returns a dict with:
        - 'success': bool
        - 'results': dict of output values
        - 'skill': the skill definition
        - 'formatted_description': text description of results
        - 'error': error message if failed
        """
        skill = self.catalog.get_skill(skill_id)
        if not skill:
            return {"success": False, "error": f"Skill not found: {skill_id}"}
        
        # Check for missing required inputs
        missing = self.catalog.get_missing_inputs(skill_id, inputs)
        if missing:
            missing_names = [m.name for m in missing]
            return {
                "success": False,
                "error": f"Missing required inputs: {missing_names}",
                "missing_inputs": [m.model_dump() for m in missing],
            }
        
        # Execute
        executor_fn = self._function_map.get(skill_id)
        if not executor_fn:
            return {
                "success": False,
                "error": f"No executor registered for skill: {skill_id}",
            }
        
        try:
            results = executor_fn(inputs)
            
            # Format description using template
            description = ""
            if skill.result_description_template:
                try:
                    template_vars = {**inputs, **results}
                    description = skill.result_description_template.format(**template_vars)
                except KeyError:
                    description = str(results)
            
            return {
                "success": True,
                "results": results,
                "skill": skill.model_dump(),
                "formatted_description": description,
                "acceptance_criteria": skill.acceptance_criteria,
            }
        except Exception as e:
            logger.error(f"Skill execution failed: {e}")
            return {"success": False, "error": str(e)}
    
    def _execute_bearing_capacity(self, inputs: dict) -> dict:

    def _execute_earth_pressure(self, inputs: dict) -> dict:
        """Execute earth pressure calculation."""
        # Import or define here
        import math
        
        phi = math.radians(inputs["backfill_friction_angle_deg"])
        H = inputs["wall_height_m"]
        gamma = inputs["backfill_unit_weight"]
        c = inputs.get("backfill_cohesion_kpa", 0.0)
        q = inputs.get("surcharge_kpa", 0.0)
        
        # Rankine coefficients
        Ka = math.tan(math.radians(45 - inputs["backfill_friction_angle_deg"] / 2)) ** 2
        Kp = math.tan(math.radians(45 + inputs["backfill_friction_angle_deg"] / 2)) ** 2
        
        # Active earth pressure at base
        sigma_a_base = Ka * gamma * H + Ka * q - 2 * c * math.sqrt(Ka)
        sigma_a_base = max(sigma_a_base, 0)
        
        # Total active force (triangular + uniform)
        Pa_soil = 0.5 * Ka * gamma * H ** 2
        Pa_surcharge = Ka * q * H
        Pa_cohesion = -2 * c * math.sqrt(Ka) * H
        
        total_Pa = max(Pa_soil + Pa_surcharge + Pa_cohesion, 0)
        
        # Point of application (from base)
        if total_Pa > 0:
            moment = Pa_soil * H / 3 + Pa_surcharge * H / 2
            y_bar = moment / total_Pa if total_Pa > 0 else H / 3
        else:
            y_bar = H / 3
        
        return {
            "Ka": round(Ka, 4),
            "Kp": round(Kp, 4),
            "sigma_a_base_kpa": round(sigma_a_base, 1),
            "total_active_force_kn_m": round(total_Pa, 1),
            "Pa_soil_kn_m": round(Pa_soil, 1),
            "Pa_surcharge_kn_m": round(Pa_surcharge, 1),
            "point_of_application_m": round(y_bar, 2),

"""
Designer Agent: Interactive geotechnical report generation.

Workflow:
1. Collect project information
2. Identify required analyses (skills)
3. Gather missing parameters
4. Execute calculations (deterministic)
5. Retrieve code references (RAG)
6. Assemble report (LLM + templates)
"""
import json
from enum import Enum
from dataclasses import dataclass, field
from loguru import logger
from src.vectordb.qdrant_store import GeoVectorStore
from src.utils.llm_client import call_llm, call_llm_with_context
from src.skills.catalog import SkillCatalog
from src.skills.executor import SkillExecutor
from src.templates.report_structure import STANDARD_REPORT_TEMPLATE
from config.settings import PRIMARY_MODEL, REASONING_MODEL


class AgentState(str, Enum):
    """States in the Designer Agent workflow."""
    INIT = "init"
    COLLECTING_PROJECT_INFO = "collecting_project_info"
    IDENTIFYING_SKILLS = "identifying_skills"
    COLLECTING_PARAMETERS = "collecting_parameters"
    EXECUTING_CALCULATIONS = "executing_calculations"
    GENERATING_REPORT = "generating_report"
    COMPLETE = "complete"


@dataclass
class ProjectData:
    """All collected project information."""
    project_name: str = ""
    project_type: str = ""  # "building", "road", "slope", "excavation", etc.
    location: str = ""
    description: str = ""
    applicable_codes: list[str] = field(default_factory=list)
    
    # Ground conditions
    soil_layers: list[dict] = field(default_factory=list)
    gwl_depth_m: float | None = None
    
    # Structure info
    foundation_type: str = ""
    loads: dict = field(default_factory=dict)
    geometry: dict = field(default_factory=dict)
    
    # Analysis parameters
    design_parameters: dict = field(default_factory=dict)
    
    # Results
    calculation_results: list[dict] = field(default_factory=list)
    retrieved_references: list[dict] = field(default_factory=list)
    
    # Raw user inputs for history
    conversation_history: list[dict] = field(default_factory=list)


class DesignerAgent:
    """
    Interactive agent that guides the user through creating
    a geotechnical design report.
    """
    
    def __init__(self, use_local_db: bool = False):
        self.store = GeoVectorStore(use_local=use_local_db)
        self.catalog = SkillCatalog()
        self.executor = SkillExecutor(self.catalog)
        self.state = AgentState.INIT
        self.project = ProjectData()
        self.selected_skills: list[str] = []
        self.current_skill_index: int = 0
        self.pending_questions: list[str] = []
    
    def start(self, initial_description: str = "") -> dict:
        """Start a new design session."""
        self.state = AgentState.COLLECTING_PROJECT_INFO
        self.project = ProjectData()
        
        if initial_description:
            return self.process_input(initial_description)
        
        return {
            "state": self.state.value,
            "message": (
                "Welcome to the Geotechnical Design Report Generator.\n\n"
                "Please describe your project. Include:\n"
                "• Project type (building foundation, slope, retaining wall, excavation)\n"
                "• Location in Hong Kong\n"
                "• Key dimensions and loads\n"
                "• Available ground investigation data\n"
                "• Applicable design codes (HK CoP, EC7, etc.)"
            ),
            "questions": [],
        }
    
    def process_input(self, user_input: str) -> dict:
        """Process user input based on current state."""
        self.project.conversation_history.append({
            "role": "user",
            "content": user_input,
        })
        
        if self.state == AgentState.COLLECTING_PROJECT_INFO:
            return self._handle_project_info(user_input)
        elif self.state == AgentState.COLLECTING_PARAMETERS:
            return self._handle_parameter_input(user_input)
        elif self.state == AgentState.IDENTIFYING_SKILLS:
            return self._identify_required_skills()
        elif self.state == AgentState.EXECUTING_CALCULATIONS:
            return self._execute_calculations()
        elif self.state == AgentState.GENERATING_REPORT:
            return self._generate_report()
        else:
            return {"state": self.state.value, "message": "Session complete."}
    
    def _handle_project_info(self, user_input: str) -> dict:
        """Extract project information from user description using LLM."""
        
        extraction_prompt = f"""Extract project information from the following description.
Return a JSON object with these fields (use null for unknown values):

{{
    "project_name": "string",
    "project_type": "one of: building, road, slope, excavation, retaining_wall, tunnel, reclamation",
    "location": "string",
    "description": "brief summary",
    "applicable_codes": ["list of code names"],
    "foundation_type": "shallow/deep/piled/not_specified",
    "soil_layers": [
        {{"depth_from_m": 0, "depth_to_m": 5, "description": "Fill", 
          "soil_type": "fill/cdg/hdg/alluvium/marine/rock",
          "spt_n": null, "cohesion_kpa": null, "friction_angle_deg": null, 
          "unit_weight_kn_m3": null}}
    ],
    "gwl_depth_m": null,
    "geometry": {{"width_m": null, "length_m": null, "depth_m": null, "height_m": null}},
    "loads": {{"permanent_kn": null, "variable_kn": null, "bearing_pressure_kpa": null}},
    "missing_critical_info": ["list of important things not mentioned"]
}}

User description:
{user_input}
"""
        
        response = call_llm(
            extraction_prompt,
            system_prompt="You are a geotechnical data extraction assistant. Return ONLY valid JSON.",
            temperature=0.1,
        )
        
        # Parse the JSON response
        try:
            # Clean up response — remove markdown code blocks if present
            cleaned = response.strip()
            if cleaned.startswith("```"):
                cleaned = cleaned.split("\n", 1)[1]
                cleaned = cleaned.rsplit("```", 1)[0]
            
            extracted = json.loads(cleaned)
        except json.JSONDecodeError:
            logger.warning("Failed to parse LLM extraction, asking for clarification")
            return {
                "state": self.state.value,
                "message": (
                    "I couldn't fully understand the project description. "
                    "Could you please provide the information in a more structured way?\n\n"
                    "For example:\n"
                    "• Project type: Building foundation\n"
                    "• Location: Kowloon\n"
                    "• Foundation: 2m x 3m pad footing at 1.5m depth\n"
                    "• Soil: 3m fill over CDG, SPT N=20-30\n"
                    "• Groundwater: 5m below ground\n"
                    "• Load: 500 kN permanent, 200 kN variable"
                ),
                "questions": [],
            }
        
        # Update project data
        self.project.project_name = extracted.get("project_name", "")
        self.project.project_type = extracted.get("project_type", "")
        self.project.location = extracted.get("location", "")
        self.project.description = extracted.get("description", "")
        self.project.applicable_codes = extracted.get("applicable_codes", [])
        self.project.foundation_type = extracted.get("foundation_type", "")
        self.project.soil_layers = extracted.get("soil_layers", [])
        self.project.gwl_depth_m = extracted.get("gwl_depth_m")
        self.project.geometry = extracted.get("geometry", {})
        self.project.loads = extracted.get("loads", {})
        
        # Generate follow-up questions for missing info
        missing = extracted.get("missing_critical_info", [])
        questions = self._generate_clarifying_questions(missing)
        
        if questions:
            self.pending_questions = questions
            self.state = AgentState.COLLECTING_PARAMETERS
            return {
                "state": self.state.value,
                "message": "I've captured the following project information:\n" + 
                          json.dumps(extracted, indent=2) +
                          "\n\nI need some additional information:",
                "questions": questions,
                "extracted_data": extracted,
            }
        else:
            # Move to skill identification
            self.state = AgentState.IDENTIFYING_SKILLS
            return self._identify_required_skills()
    
    def _generate_clarifying_questions(self, missing_items: list[str]) -> list[str]:
        """Generate specific questions for missing information."""
        standard_questions = {
            "soil_parameters": (
                "What are the soil parameters for the main bearing stratum? "
                "(cohesion c', friction angle φ', unit weight γ)"
            ),
            "groundwater": (
                "What is the groundwater level depth? "
                "(metres below ground surface)"
            ),
            "loads": (
                "What are the characteristic loads? "
                "(permanent/dead load in kN, variable/live load in kN)"
            ),
            "geometry": (
                "What are the foundation dimensions? "
                "(width, length, embedment depth in metres)"
            ),
            "design_code": (
                "Which design approach should be used? "
                "(Traditional with global FoS / EC7 Design Approach 1)"
            ),
            "slope_geometry": (
                "What is the slope geometry? "
                "(height in metres, slope angle in degrees)"
            ),
        }
        
        questions = []
        for item in missing_items:
            item_lower = item.lower()
            for key, question in standard_questions.items():
                if key in item_lower or any(
                    word in item_lower for word in key.split("_")
                ):
                    questions.append(question)
                    break
            else:
                questions.append(f"Please provide: {item}")
        
        return questions[:5]  # Limit to 5 questions at a time
    
    def _handle_parameter_input(self, user_input: str) -> dict:
        """Process answers to clarifying questions."""
        # Use LLM to extract parameters from the answer
        extraction_prompt = f"""The user was asked these questions:
{json.dumps(self.pending_questions, indent=2)}

Their response:
{user_input}

Extract any numerical values and parameters mentioned. 
Return a JSON object with the values found.
Use keys like: cohesion_kpa, friction_angle_deg, unit_weight_kn_m3, 
gwl_depth_m, foundation_width_m, foundation_length_m, foundation_depth_m,
permanent_load_kn, variable_load_kn, slope_height_m, slope_angle_deg, etc.
Only include values that were actually provided. Return ONLY valid JSON.
"""
        
        response = call_llm(
            extraction_prompt,
            system_prompt="You are a parameter extraction assistant. Return ONLY valid JSON.",
            temperature=0.1,
        )
        
        try:
            cleaned = response.strip()
            if cleaned.startswith("```"):
                cleaned = cleaned.split("\n", 1)[1]
                cleaned = cleaned.rsplit("```", 1)[0]
            params = json.loads(cleaned)
        except json.JSONDecodeError:
            params = {}
        
        # Update project data
        self.project.design_parameters.update(params)
        
        if params.get("gwl_depth_m") is not None:
            self.project.gwl_depth_m = params["gwl_depth_m"]
        
        # Check if we have enough info now
        self.state = AgentState.IDENTIFYING_SKILLS
        return self._identify_required_skills()
    
    def _identify_required_skills(self) -> dict:
        """Determine which skills/analyses are needed for this project."""
        
        project_summary = json.dumps({
            "type": self.project.project_type,
            "description": self.project.description,
            "foundation_type": self.project.foundation_type,
            "geometry": self.project.geometry,
            "parameters": self.project.design_parameters,
        }, indent=2)
        
        available_skills = json.dumps(self.catalog.list_all(), indent=2)
        
        skill_selection_prompt = f"""Given this project:
{project_summary}

And these available analysis skills:
{available_skills}

Which skills should be applied? Return a JSON array of skill_id values.
Consider:
- Foundation projects need: bearing capacity, settlement
- Slope projects need: slope stability
- Retaining wall projects need: earth pressure, bearing capacity, sliding check
- Excavation projects need: earth pressure, base heave, ground movement

Return ONLY a JSON array, e.g.: ["shallow_bearing_capacity", "slope_stability_infinite"]
"""
        
        response = call_llm(
            skill_selection_prompt,
            system_prompt="You are a geotechnical engineering assistant. Return ONLY a JSON array.",
            temperature=0.1,
        )
        
        try:
            cleaned = response.strip()
            if cleaned.startswith("```"):
                cleaned = cleaned.split("\n", 1)[1]
                cleaned = cleaned.rsplit("```", 1)[0]
            self.selected_skills = json.loads(cleaned)
        except json.JSONDecodeError:
            # Default based on project type
            type_skill_map = {
                "building": ["shallow_bearing_capacity"],
                "slope": ["slope_stability_infinite"],
                "retaining_wall": ["earth_pressure_rankine", "shallow_bearing_capacity"],
                "excavation": ["earth_pressure_rankine"],
            }
            self.selected_skills = type_skill_map.get(
                self.project.project_type, ["shallow_bearing_capacity"]
            )
        
        logger.info(f"Selected skills: {self.selected_skills}")
        
        # Check if we have all needed inputs for selected skills
        all_missing = []
        for skill_id in self.selected_skills:
            skill = self.catalog.get_skill(skill_id)
            if skill:
                for inp in skill.inputs:
                    if inp.required and inp.name not in self.project.design_parameters:
                        # Try to find in project data
                        value = self._find_parameter_value(inp.name)
                        if value is not None:
                            self.project.design_parameters[inp.name] = value
                        else:
                            all_missing.append(inp)
        
        if all_missing:
            # Need more info — ask user
            questions = [
                f"{inp.description} ({inp.name}, unit: {inp.unit})"
                for inp in all_missing
            ]
            # Deduplicate
            questions = list(dict.fromkeys(questions))
            self.pending_questions = questions
            self.state = AgentState.COLLECTING_PARAMETERS
            
            return {
                "state": self.state.value,
                "message": (
                    f"I've identified the following analyses needed:\n"
                    + "\n".join(f"  • {self.catalog.get_skill(s).name}" for s in self.selected_skills if self.catalog.get_skill(s))
                    + "\n\nBut I need these additional parameters:"
                ),
                "questions": questions[:8],
                "selected_skills": self.selected_skills,
            }
        
        # We have everything — execute
        self.state = AgentState.EXECUTING_CALCULATIONS
        return self._execute_calculations()
    
    def _find_parameter_value(self, param_name: str):
        """Try to find a parameter value from existing project data."""
        # Direct match in design_parameters
        if param_name in self.project.design_parameters:
            return self.project.design_parameters[param_name]
        
        # Map from project geometry/loads
        geometry_map = {
            "foundation_width": self.project.geometry.get("width_m"),
            "foundation_length": self.project.geometry.get("length_m"),
            "foundation_depth": self.project.geometry.get("depth_m"),
            "wall_height_m": self.project.geometry.get("height_m"),
            "slope_angle_deg": self.project.geometry.get("slope_angle_deg"),
        }
        if param_name in geometry_map:
            return geometry_map[param_name]
        
        # Map from soil layers (use first layer as default)
        if self.project.soil_layers:
            soil = self.project.soil_layers[0]
            soil_map = {
                "soil_cohesion_kpa": soil.get("cohesion_kpa"),
                "soil_friction_angle_deg": soil.get("friction_angle_deg"),
                "soil_unit_weight": soil.get("unit_weight_kn_m3"),
                "cohesion_kpa": soil.get("cohesion_kpa"),
                "friction_angle_deg": soil.get("friction_angle_deg"),
                "unit_weight": soil.get("unit_weight_kn_m3"),
                "backfill_friction_angle_deg": soil.get("friction_angle_deg"),
                "backfill_unit_weight": soil.get("unit_weight_kn_m3"),
            }
            if param_name in soil_map:
                return soil_map[param_name]
        
        # GWL
        if param_name == "gwl_depth" and self.project.gwl_depth_m is not None:
            return self.project.gwl_depth_m
        
        return None
    
    def _execute_calculations(self) -> dict:
        """Execute all selected skills."""
        results = []
        
        for skill_id in self.selected_skills:
            logger.info(f"Executing skill: {skill_id}")
            result = self.executor.execute(skill_id, self.project.design_parameters)
            results.append({
                "skill_id": skill_id,
                "result": result,
            })
            
            if result["success"]:
                logger.info(f"  ✓ {skill_id} completed successfully")
            else:
                logger.warning(f"  ✗ {skill_id} failed: {result.get('error')}")
        
        self.project.calculation_results = results
        
        # Move to report generation
        self.state = AgentState.GENERATING_REPORT
        return self._generate_report()
    
    def _generate_report(self) -> dict:
        """Generate the full report using LLM + templates + calculation results."""
        
        # Step 1: Retrieve relevant code references
        logger.info("Retrieving relevant code references...")
        for skill_id in self.selected_skills:
            skill = self.catalog.get_skill(skill_id)
            if skill:
                for ref in skill.code_references:
                    chunks = self.store.search(
                        query=f"{ref.description} {ref.clause}",
                        top_k=3,
                        document_type="code",
                    )
                    self.project.retrieved_references.extend(chunks)
        
        # Step 2: Generate each report section
        report_sections = []
        
        for section_template in STANDARD_REPORT_TEMPLATE:
            logger.info(f"Generating section: {section_template.title}")
            
            section_content = self._generate_section(section_template)
            report_sections.append({
                "number": section_template.number,
                "title": section_template.title,
                "content": section_content,
            })
        
        # Step 3: Assemble final report
        report_markdown = self._assemble_report(report_sections)
        
        self.state = AgentState.COMPLETE
        
        return {
            "state": self.state.value,
            "message": "Report generation complete.",
            "report": report_markdown,
            "calculation_results": self.project.calculation_results,
            "references_used": [
                {
                    "document": r["metadata"].get("document_name"),
                    "section": r["metadata"].get("section_title"),
                }
                for r in self.project.retrieved_references[:20]
            ],
        }

GUIDANCE: {template.guidance}

"""
Validator Agent: Checks third-party geotechnical submissions 
for completeness and compliance.

Levels of checking:
  Level 1: Structural completeness (are all required sections present?)
  Level 2: Parameter extraction and reasonableness
  Level 3: Code compliance checking (against vector DB)
"""
import json
from dataclasses import dataclass, field
from loguru import logger
from src.ingestion.pdf_processor import extract_text_from_pdf, extract_pages_with_numbers
from src.vectordb.qdrant_store import GeoVectorStore
from src.utils.llm_client import call_llm, call_llm_with_context
from src.templates.report_structure import STANDARD_REPORT_TEMPLATE
from config.settings import PRIMARY_MODEL, REASONING_MODEL


@dataclass
class CheckItem:
    """A single compliance check result."""
    category: str  # "completeness", "parameter", "compliance", "consistency"
    item: str  # What was checked
    status: str  # "pass", "fail", "warning", "not_checked"
    details: str = ""
    reference: str = ""  # Code clause reference
    severity: str = "medium"  # "critical", "high", "medium", "low"


@dataclass
class ValidationReport:
    """Complete validation report."""
    document_name: str
    checks: list[CheckItem] = field(default_factory=list)
    summary: str = ""
    overall_status: str = ""  # "acceptable", "revisions_needed", "rejected"
    
    @property
    def pass_count(self) -> int:
        return sum(1 for c in self.checks if c.status == "pass")
    
    @property
    def fail_count(self) -> int:
        return sum(1 for c in self.checks if c.status == "fail")
    
    @property
    def warning_count(self) -> int:
        return sum(1 for c in self.checks if c.status == "warning")


class ValidatorAgent:
    """
    Validates third-party geotechnical reports against
    HK codes and standard practice.
    """
    
    def __init__(self, use_local_db: bool = False):
        self.store = GeoVectorStore(use_local=use_local_db)
    
    def validate_report(self, pdf_path: str) -> ValidationReport:
        """
        Full validation pipeline for a third-party report.
        
        Args:
            pdf_path: Path to the PDF report to validate
            
        Returns:
            ValidationReport with all check results
        """
        document_name = pdf_path.split("/")[-1]
        report = ValidationReport(document_name=document_name)
        
        # Step 1: Extract text
        logger.info(f"Step 1: Extracting text from {pdf_path}")
        full_text = extract_text_from_pdf(pdf_path)
        pages = extract_pages_with_numbers(pdf_path)
        
        if not full_text or len(full_text.strip()) < 200:
            report.checks.append(CheckItem(
                category="extraction",
                item="Document readability",
                status="fail",
                details="Could not extract meaningful text from the document. "
                        "The PDF may be scanned/image-based.",
                severity="critical",
            ))
            report.overall_status = "rejected"
            return report
        
        # Step 2: Level 1 — Structural completeness check
        logger.info("Step 2: Checking report structure completeness")
        completeness_checks = self._check_completeness(full_text)
        report.checks.extend(completeness_checks)
        
        # Step 3: Level 2 — Extract and check design parameters
        logger.info("Step 3: Extracting and checking design parameters")
        parameter_checks = self._check_parameters(full_text)
        report.checks.extend(parameter_checks)
        
        # Step 4: Level 3 — Code compliance checking
        logger.info("Step 4: Checking code compliance")
        compliance_checks = self._check_code_compliance(full_text)
        report.checks.extend(compliance_checks)
        
        # Step 5: Internal consistency
        logger.info("Step 5: Checking internal consistency")
        consistency_checks = self._check_consistency(full_text)
        report.checks.extend(consistency_checks)
        
        # Generate summary
        report.summary = self._generate_summary(report)
        
        # Determine overall status
        critical_fails = sum(
            1 for c in report.checks
            if c.status == "fail" and c.severity == "critical"
        )
        if critical_fails > 0:
            report.overall_status = "rejected"
        elif report.fail_count > 3:
            report.overall_status = "revisions_needed"
        elif report.fail_count > 0:
            report.overall_status = "revisions_needed"
        else:
            report.overall_status = "acceptable"
        
        return report
    
    def _check_completeness(self, text: str) -> list[CheckItem]:
        """Level 1: Check if all required report sections are present."""
        
        expected_sections = [t for t in STANDARD_REPORT_TEMPLATE if t.required]
        
        prompt = f"""Analyse this geotechnical report and determine which standard sections 
are present or absent.

Expected sections:
{json.dumps([{"number": s.number, "title": s.title} for s in expected_sections], indent=2)}

Report text (first 8000 characters):
{text[:8000]}

For each expected section, determine:
- "present": the section exists and has substantive content
- "absent": the section is missing entirely
- "inadequate": the section exists but has very little content

Return a JSON array:
[{{"section": "title", "status": "present/absent/inadequate", "notes": "brief note"}}]

Return ONLY valid JSON.
"""
        
        response = call_llm(prompt, temperature=0.1, max_tokens=2000)
        
        checks = []
        try:
            cleaned = response.strip()
            if cleaned.startswith("```"):
                cleaned = cleaned.split("\n", 1)[1]
                cleaned = cleaned.rsplit("```", 1)[0]
            section_results = json.loads(cleaned)
            
            for result in section_results:
                status = "pass" if result["status"] == "present" else "fail"
                if result["status"] == "inadequate":
                    status = "warning"
                
                checks.append(CheckItem(
                    category="completeness",
                    item=f"Section: {result['section']}",
                    status=status,
                    details=result.get("notes", ""),
                    severity="high" if result["status"] == "absent" else "medium",
                ))
        except json.JSONDecodeError:
            checks.append(CheckItem(
                category="completeness",
                item="Section analysis",
                status="not_checked",
                details="Could not parse section analysis results",
            ))
        
        return checks
    
    def _check_parameters(self, text: str) -> list[CheckItem]:
        """Level 2: Extract design parameters and check reasonableness."""
        
        prompt = f"""Extract all geotechnical design parameters from this report.
Look for:
- Soil parameters (cohesion c', friction angle φ', unit weight γ, SPT N values)
- Groundwater level
- Foundation dimensions and depths
- Design loads
- Factors of safety used
- Bearing capacity values
- Settlement estimates

Report text (first 10000 characters):
{text[:10000]}

Return a JSON object:
{{
    "soil_parameters": [
        {{"layer": "description", "depth_m": "range", "c_kpa": null, "phi_deg": null, 
          "gamma_kn_m3": null, "spt_n": null, "cu_kpa": null}}
    ],
    "gwl_depth_m": null,
    "foundation": {{"type": "", "width_m": null, "depth_m": null}},
    "factors_of_safety": {{"bearing": null, "sliding": null, "overturning": null, "slope": null}},
    "design_method": "",
    "bearing_capacity_kpa": null,
    "settlement_mm": null,
    "issues_noticed": ["list of any obvious issues"]
}}

Return ONLY valid JSON.
"""
        
        response = call_llm(prompt, temperature=0.1, max_tokens=2000)
        
        checks = []
        try:
            cleaned = response.strip()
            if cleaned.startswith("```"):
                cleaned = cleaned.split("\n", 1)[1]
                cleaned = cleaned.rsplit("```", 1)[0]
            params = json.loads(cleaned)
            
            # Check soil parameter reasonableness
            for layer in params.get("soil_parameters", []):
                # Friction angle checks
                phi = layer.get("phi_deg")
                if phi is not None:
                    if phi < 15 or phi > 45:
                        checks.append(CheckItem(
                            category="parameter",
                            item=f"Friction angle for {layer.get('layer', 'unknown layer')}",
                            status="warning",
                            details=f"φ' = {phi}° is outside typical range (20-40° for HK soils)",
                            severity="high",
                        ))
                    else:
                        checks.append(CheckItem(
                            category="parameter",
                            item=f"Friction angle for {layer.get('layer', 'unknown layer')}",
                            status="pass",
                            details=f"φ' = {phi}° is within typical range",
                        ))
                
                # Unit weight checks
                gamma = layer.get("gamma_kn_m3")
                if gamma is not None:
                    if gamma < 14 or gamma > 25:
                        checks.append(CheckItem(
                            category="parameter",
                            item=f"Unit weight for {layer.get('layer', 'unknown layer')}",
                            status="warning",
                            details=f"γ = {gamma} kN/m³ is outside typical range (16-22 kN/m³)",
                            severity="medium",
                        ))
            
            # Check factors of safety
            fos_requirements = {
                "bearing": {"min": 3.0, "ref": "Code of Practice for Foundations 2017"},
                "sliding": {"min": 1.5, "ref": "Geoguide 1"},
                "overturning": {"min": 2.0, "ref": "Geoguide 1"},
                "slope": {"min": 1.4, "ref": "Geotechnical Manual for Slopes"},
            }
            
            for fos_type, requirement in fos_requirements.items():
                fos_value = params.get("factors_of_safety", {}).get(fos_type)
                if fos_value is not None:
                    if fos_value < requirement["min"]:
                        checks.append(CheckItem(
                            category="parameter",
                            item=f"Factor of safety ({fos_type})",
                            status="fail",
                            details=(
                                f"FoS = {fos_value} is below minimum requirement "
                                f"of {requirement['min']}"
                            ),
                            reference=requirement["ref"],
                            severity="critical",
                        ))
                    else:
                        checks.append(CheckItem(
                            category="parameter",
                            item=f"Factor of safety ({fos_type})",
                            status="pass",
                            details=f"FoS = {fos_value} ≥ {requirement['min']}",
                            reference=requirement["ref"],
                        ))
            
            # Record any issues noticed
            for issue in params.get("issues_noticed", []):
                checks.append(CheckItem(
                    category="parameter",
                    item="Noticed issue",
                    status="warning",
                    details=issue,
                    severity="medium",
                ))
                
        except json.JSONDecodeError:
            checks.append(CheckItem(
                category="parameter",
                item="Parameter extraction",
                status="not_checked",
                details="Could not parse parameter extraction results",
            ))
        
        return checks
    
    def _check_code_compliance(self, text: str) -> list[CheckItem]:
        """Level 3: Check design methods against codes in vector DB."""
        checks = []
        
        # Extract the design method(s) used
        method_prompt = f"""From this geotechnical report, identify:
1. What design methods/equations were used (e.g., Terzaghi, Meyerhof, Hansen, Bishop)
2. What codes/standards are referenced
3. What limit state design approach is used (if any)

Report excerpt:
{text[:6000]}

Return a JSON object:
{{
    "design_methods": ["list of methods mentioned"],
    "referenced_codes": ["list of codes/standards"],
    "limit_state_approach": "description or null"
}}

Return ONLY valid JSON.
"""
        
        response = call_llm(method_prompt, temperature=0.1, max_tokens=1000)
        
        try:
            cleaned = response.strip()
            if cleaned.startswith("```"):
                cleaned = cleaned.split("\n", 1)[1]
                cleaned = cleaned.rsplit("```", 1)[0]
            methods = json.loads(cleaned)
            
            # Check each method against our code database
            for method in methods.get("design_methods", []):
                # Search vector DB for relevant code provisions
                code_chunks = self.store.search(
                    query=f"design method {method} requirements",
                    top_k=3,
                    document_type="code",
                )
                
                if code_chunks:
                    # Ask LLM to compare the method against code requirements
                    comparison_prompt = f"""The report uses the "{method}" method.

The relevant code provisions state:
{json.dumps([c['text'][:500] for c in code_chunks], indent=2)}

Is this method acceptable per the codes? Any specific requirements that must be met?
Answer briefly as JSON:
{{"acceptable": true/false, "notes": "brief explanation", "clause": "relevant clause"}}
"""
                    comp_response = call_llm(comparison_prompt, temperature=0.1, max_tokens=500)
                    
                    try:
                        comp_cleaned = comp_response.strip()
                        if comp_cleaned.startswith("```"):
                            comp_cleaned = comp_cleaned.split("\n", 1)[1]
                            comp_cleaned = comp_cleaned.rsplit("```", 1)[0]
                        comp_result = json.loads(comp_cleaned)
                        
                        checks.append(CheckItem(
                            category="compliance",
                            item=f"Design method: {method}",
                            status="pass" if comp_result.get("acceptable") else "warning",
                            details=comp_result.get("notes", ""),
                            reference=comp_result.get("clause", ""),
                        ))
                    except json.JSONDecodeError:
                        pass
            
            # Check if required codes are referenced
            required_hk_codes = [
                "Code of Practice for Foundations",
                "Geoguide",
                "BS EN 1997",
                "Eurocode 7",
            ]
            referenced = methods.get("referenced_codes", [])
            referenced_lower = " ".join(referenced).lower()
            
            for req_code in required_hk_codes:
                if req_code.lower() in referenced_lower:
                    checks.append(CheckItem(
                        category="compliance",
                        item=f"Reference to {req_code}",
                        status="pass",
                        details=f"{req_code} is referenced in the report",
                    ))
            
        except json.JSONDecodeError:
            pass
        
        return checks
    
    def _check_consistency(self, text: str) -> list[CheckItem]:
        """Check internal consistency of the report."""
        
        prompt = f"""Review this geotechnical report for internal consistency issues.
Check for:
1. Soil parameters used in calculations matching those described in the ground conditions section
2. Design loads matching structural information
3. Recommended foundation type matching the analysis performed
4. Groundwater level used consistently throughout
5. Any contradictions between different sections

Report text:
{text[:12000]}

Return a JSON array of any inconsistencies found:
[{{"issue": "description", "severity": "high/medium/low", "sections_involved": "which sections"}}]

If no inconsistencies are found, return an empty array: []
Return ONLY valid JSON.
"""
        
        response = call_llm(prompt, temperature=0.2, max_tokens=1500, model=REASONING_MODEL)
        
        checks = []
        try:
            cleaned = response.strip()
            if cleaned.startswith("```"):
                cleaned = cleaned.split("\n", 1)[1]
                cleaned = cleaned.rsplit("```", 1)[0]
            issues = json.loads(cleaned)
            
            for issue in issues:
                checks.append(CheckItem(
                    category="consistency",
                    item=issue.get("issue", "Consistency issue"),
                    status="warning" if issue.get("severity") != "high" else "fail",
                    details=f"Sections involved: {issue.get('sections_involved', 'N/A')}",
                    severity=issue.get("severity", "medium"),
                ))
            
            if not issues:
                checks.append(CheckItem(
                    category="consistency",
                    item="Internal consistency",
                    status="pass",
                    details="No obvious inconsistencies detected",
                ))
                
        except json.JSONDecodeError:
            pass
        
        return checks
    
    def _generate_summary(self, report: ValidationReport) -> str:
        """Generate a human-readable summary of the validation."""
        
        summary_data = {
            "total_checks": len(report.checks),
            "passed": report.pass_count,
            "failed": report.fail_count,
            "warnings": report.warning_count,
            "critical_issues": [
                c.item for c in report.checks
                if c.status == "fail" and c.severity == "critical"
            ],
            "all_issues": [
                {"item": c.item, "status": c.status, "details": c.details}
                for c in report.checks if c.status in ("fail", "warning")
            ],
        }
        
        prompt = f"""Write a brief professional summary of this geotechnical report review.

Review results:
{json.dumps(summary_data, indent=2)}

Write 2-3 paragraphs summarising:
1. Overall assessment
2. Key issues found
3. Recommended actions for the submitting engineer

Write in a formal, professional tone appropriate for correspondence 
between government engineers in Hong Kong.
"""
        
        return call_llm(prompt, temperature=0.3, max_tokens=800)
    
    def format_validation_report(self, report: ValidationReport) -> str:
        """Format the validation report as markdown."""
        lines = [
            f"# SUBMISSION REVIEW REPORT",
            f"",
            f"**Document Reviewed:** {report.document_name}",
            f"**Overall Status:** {report.overall_status.upper().replace('_', ' ')}",
            f"**Checks Performed:** {len(report.checks)} "
            f"(✓ {report.pass_count} | ✗ {report.fail_count} | ⚠ {report.warning_count})",
            f"",
            f"---",
            f"",
            f"## Summary",
            f"",
            report.summary,
            f"",
            f"---",
            f"",
            f"## Detailed Findings",
            f"",
        ]
        
        # Group checks by category
        categories = {}
        for check in report.checks:
            if check.category not in categories:
                categories[check.category] = []
            categories[check.category].append(check)
        
        status_emoji = {
            "pass": "✅",
            "fail": "❌",
            "warning": "⚠️",
            "not_checked": "⬜",
        }
        
        for category, checks in categories.items():
            lines.append(f"### {category.replace('_', ' ').title()}")
            lines.append("")
            lines.append("| Status | Item | Details | Reference |")
            lines.append("|--------|------|---------|-----------|")
            
            for check in checks:
                emoji = status_emoji.get(check.status, "⬜")
                details = check.details[:100] + "..." if len(check.details) > 100 else check.details
                lines.append(
                    f"| {emoji} {check.status.upper()} | {check.item} | "
                    f"{details} | {check.reference} |"
                )
            
            lines.append("")
        
        lines.extend([
            "---",
            "",
            "## Recommended Actions",
            "",
            "Items marked ❌ must be addressed before the submission can be accepted.",
            "Items marked ⚠️ should be reviewed and clarified.",
            "",
            "*This review was generated with AI assistance and should be verified by "
            "the reviewing engineer before issue.*",
        ])
        
        return "\n".join(lines)

            st.session_state.designer_messages.append({
                "role": "user", "content": prompt
            })
            
            with st.spinner("Processing..."):
                result = st.session_state.designer_agent.process_input(prompt)
            
            message_data = {
                "role": "assistant",
                "content": result.get("message", ""),
                "questions": result.get("questions", []),
            }
            
            if result.get("report"):
                message_data["report"] = result["report"]
            
            st.session_state.designer_messages.append(message_data)
            st.rerun()

                tmp.write(uploaded_file.getbuffer())
                tmp_path = tmp.name

                with st.spinner("Analysing report... This may take 2-5 minutes."):
                    report = st.session_state.validator_agent.validate_report(tmp_path)
                    formatted = st.session_state.validator_agent.format_validation_report(report)
                
                # Display results
                # Status banner
                status_colors = {
                    "acceptable": "green",
                    "revisions_needed": "orange",
                    "rejected": "red",
                }
                status_text = report.overall_status.upper().replace("_", " ")
                
                col1, col2, col3, col4 = st.columns(4)
                col1.metric("Overall Status", status_text)
                col2.metric("✅ Passed", report.pass_count)
                col3.metric("❌ Failed", report.fail_count)
                col4.metric("⚠️ Warnings", report.warning_count)
                
                st.divider()
                
                # Full report
                st.markdown(formatted)
                
                # Download
                st.download_button(
                    "📥 Download Review Report",
                    formatted,
                    file_name=f"review_{uploaded_file.name.replace('.pdf', '.md')}",
                    mime="text/markdown",
                )
            finally:
                os.unlink(tmp_path)


# ─── Mode 4: Document Manager ───
elif mode == "📂 Document Manager":
    st.header("📂 Knowledge Base Manager")
    st.caption("Manage the documents in your vector database.")
    
    # List current documents
    st.subheader("Current Documents")
    docs = st.session_state.qa_agent.list_knowledge_base()
    
    if docs:
        for doc in docs:
            col1, col2, col3 = st.columns([3, 1, 1])
            col1.write(f"📄 **{doc['document_name']}**")
            col2.write(f"Type: {doc['document_type']}")
            col3.write(f"Chunks: {doc['chunk_count']}")
    else:
        st.info("No documents ingested yet. Upload your first document below.")
    
    st.divider()
    
    # Upload new document
    st.subheader("Add New Document")
    
    uploaded = st.file_uploader(
        "Upload a PDF document",
        type=["pdf"],
        key="ingest_upload",
    )
    
    if uploaded:
        col1, col2 = st.columns(2)
        with col1:
            doc_name = st.text_input(
                "Document Name",
                value=uploaded.name.replace(".pdf", ""),
            )
            doc_id = st.text_input(
                "Document ID",
                value=uploaded.name.replace(".pdf", "").lower().replace(" ", "_"),
            )
        with col2:
            doc_type = st.selectbox(
                "Document Type",
                ["code", "manual", "guideline", "report", "drawing"],
            )
        
        if st.button("📥 Ingest Document", type="primary"):
            with tempfile.NamedTemporaryFile(delete=False, suffix=".pdf") as tmp:
                tmp.write(uploaded.getbuffer())
                tmp_path = tmp.name
            
            try:
                with st.spinner(f"Processing {doc_name}... This may take a few minutes."):
                    num_chunks = ingest_document(
                        pdf_path=tmp_path,
                        document_id=doc_id,
                        document_name=doc_name,
                        document_type=doc_type,
                        use_local_db=True,
                    )
                
                st.success(f"✅ Successfully ingested '{doc_name}' ({num_chunks} chunks)")
                st.rerun()
            except Exception as e:
                st.error(f"❌ Error ingesting document: {e}")
            finally:
                os.unlink(tmp_path)

  □ Run bearing capacity with known inputs → verify against hand calc
  □ Run slope stability → verify against hand calc
  □ Run earth pressure → verify against hand calc

□ 4. DESIGNER AGENT

  □ Answer clarifying questions → should execute calculations
  □ Review generated report → check structure and accuracy
  □ Verify code citations are real (not hallucinated)

□ 5. VALIDATOR AGENT

  □ Upload a well-structured report → are passes reasonable?
  □ Upload a poorly structured report → are failures appropriate?
Step 6.2 — Common Issues and Fixes
Issue	Likely Cause	Fix
Empty search results	Embedding model not loaded or chunks too small	Check embed_query has the BGE instruction prefix; increase chunk size
LLM returns non-JSON	Model doesn't follow structured output instructions	Add "Return ONLY valid JSON, no markdown" to prompts; use regex fallback parsing
Wrong code citations	LLM hallucinating clause numbers	Post-process: verify every citation against vector DB search
Slow performance	Large model or too many LLM calls	Use lighter model for simple tasks; cache embeddings; batch LLM calls
PDF extraction fails	Scanned/image PDF	Use pytesseract OCR pipeline; or manually convert problematic PDFs
Calculation errors	Bug in deterministic code	Write unit tests; compare against textbook examples
