Step 1: Query Understanding & Routing

def route_design_query(query: str) -> Dict:
    """
    Classify query and determine retrieval strategy
    """
    query_lower = query.lower()
    
    # Detect query type
    if any(term in query_lower for term in ["how to design", "design procedure", "steps for"]):
        return {
            "type": "comprehensive_design",
            "strategy": "hierarchical_graph",
            "target_elements": extract_elements(query),  # ["beam", "concrete"]
            "target_actions": extract_actions(query)      # ["design", "calculate"]
        }
    elif any(term in query_lower for term in ["what is", "define", "meaning of"]):
        return {"type": "definition", "strategy": "atomic_lookup"}
    elif any(term in query_lower for term in ["table", "value", "coefficient"]):
        return {"type": "data_lookup", "strategy": "table_first"}
    
    return {"type": "general", "strategy": "semantic_search"}

Step 2: Hierarchical + Semantic Retrieval

def retrieve_design_components(query: str, route: Dict, vector_store, k=5):
    """
    Multi-stage retrieval for comprehensive design queries
    """
    results = []
    
    # Stage 1: Find relevant SECTIONS (broad context)
    if route["type"] == "comprehensive_design":
        # Search section overview index
        section_chunks = vector_store["section_index"].similarity_search(
            query, 
            k=3,
            filter={"content_type": "section_overview"}
        )
        relevant_clauses = [c.metadata["clause_id"] for c in section_chunks]
        
        # Stage 2: Find specific DESIGN RULES within those sections
        design_chunks = vector_store["rule_index"].similarity_search(
            query,
            k=k,
            filter={"clause_id": {"$in": relevant_clauses}}  # Hierarchical filter
        )
        results.extend(design_chunks)
    
    # Stage 3: Fallback - direct semantic search for missed items
    if len(results) < k:
        fallback = vector_store["rule_index"].similarity_search(
            query, k=k - len(results)
        )
        results.extend(fallback)
    
    return deduplicate_by_clause(results)

Step 3: Graph Expansion (Fetch Dependencies)

def expand_with_dependencies(chunks: List, vector_store, max_depth=1):
    """
    Fetch referenced tables, equations, definitions dynamically
    """
    expanded = []
    fetched_ids = set()
    
    for chunk in chunks:
        # Keep original chunk with its canonical source
        expanded.append({
            "content": chunk.text,
            "source": chunk.metadata["canonical_source"],
            "role": "primary"
        })
        fetched_ids.add(chunk.metadata["chunk_id"])
        
        # Fetch referenced components (tables, equations, definitions)
        for ref_id in chunk.metadata.get("dependencies", {}).get("references", []):
            if ref_id in fetched_ids:
                continue
                
            ref_chunk = vector_store.get_by_id(ref_id)
            if ref_chunk:
                expanded.append({
                    "content": ref_chunk.text,
                    "source": ref_chunk.metadata["canonical_source"],  # Original location!
                    "role": "reference",
                    "referenced_from": chunk.metadata["canonical_source"]["clause_id"]
                })
                fetched_ids.add(ref_id)
    
    return expanded

Step 4: Citation-Aware Context Assembly

def assemble_llm_context(retrieved_items: List[Dict], query: str) -> str:
    """
    Format context with clear source attribution for the LLM
    """
    context = f"Query: {query}\n\n"
    context += "## Retrieved Design Information\n\n"
    
    # Group by source clause for organized output
    by_clause = {}
    for item in retrieved_items:
        clause = item["source"]["clause_id"]
        if clause not in by_clause:
            by_clause[clause] = {
                "title": item["source"]["clause_title"],
                "page": item["source"]["page_number"],
                "items": []
            }
        by_clause[clause]["items"].append(item)
    
    # Format each clause group
    for clause_id, group in by_clause.items():
        context += f"### [{clause_id}] {group['title']} (Page {group['page']})\n"
        
        for item in group["items"]:
            if item["role"] == "primary":
                context += f"{item['content']}\n\n"
            elif item["role"] == "reference":
                # Clearly mark referenced content with ITS original source
                ref_clause = item["source"]["clause_id"]
                context += f"> *Referenced from {ref_clause}:*\n"
                context += f"> {item['content']}\n\n"
    
    # Add citation instructions
    context += """
## Citation Instructions
- When stating a design rule, cite the clause where it appears: [Clause X.Y.Z]
- When using a table value, cite the table's original clause, not where it was referenced
- When quoting an equation, include its equation number and source clause
- Include page numbers for PDF verification: (p. XX)
"""
    
    return context

from pydantic import BaseModel, Field
from typing import List, Optional, Dict, Any
from enum import Enum

class ContentType(str, Enum):
    SECTION_OVERVIEW = "section_overview"
    DESIGN_RULE = "design_rule"
    TABLE = "table"
    EQUATION = "equation"
    DEFINITION = "definition"

class CanonicalSource(BaseModel):
    """Immutable source information for citation integrity"""
    clause_id: str
    clause_title: str
    page_number: int
    document_id: str = "CoP_SUC2013e"

class ChunkMetadata(BaseModel):
    chunk_id: str
    content_type: ContentType
    canonical_source: CanonicalSource
    
    # Graph Links
    references: List[str] = Field(default_factory=list)  # IDs of referenced chunks
    referenced_by: List[str] = Field(default_factory=list)
    
    # Engineering Specifics
    regulatory_strength: Optional[str] = None  # "mandatory", "advisory"
    table_caption: Optional[str] = None
    equation_id: Optional[str] = None

class DesignChunk(BaseModel):
    """The fundamental unit of knowledge"""
    id: str
    text: str
    meta ChunkMetadata
    embedding: Optional[List[float]] = None

2. parser.py (Docling & Logical Assembly)
Handles structure-aware parsing and groups items into logical design concepts.

from docling.document_converter import DocumentConverter
from models import DesignChunk, ChunkMetadata, CanonicalSource, ContentType
import re
import uuid

class EngineeringParser:
    def __init__(self):
        self.converter = DocumentConverter()
        # Regex for Clause IDs (e.g., "6.1.2", "Table 6.3")
        self.clause_pattern = re.compile(r'^(\d+(?:\.\d+)*)')
        self.table_pattern = re.compile(r'Table\s+(\d+(?:\.\d+)*)', re.I)
        self.clause_ref_pattern = re.compile(r'Clause\s+(\d+(?:\.\d+)*)', re.I)

    def parse_pdf(self, pdf_path: str) -> List[DesignChunk]:
        result = self.converter.convert(pdf_path)
        doc = result.document
        return self.assemble_logical_chunks(doc)

    def assemble_logical_chunks(self, doc) -> List[DesignChunk]:
        chunks = []
        current_content = []
        current_meta = None
        
        for item, level in doc.iterate_items():
            item_type = item.label.name
            
            # 1. Handle Section Headers (Start New Logical Concept)
            if item_type == "SECTION_HEADER":
                if current_content:
                    chunks.append(self._create_chunk(current_content, current_meta))
                    current_content = []
                
                clause_id = self._extract_clause_id(item.text)
                current_meta = {
                    "clause_id": clause_id,
                    "title": item.text,
                    "page": item.page_no,
                    "type": ContentType.DESIGN_RULE
                }
                current_content.append(f"# {item.text}")
            
            # 2. Handle Tables (Atomic Units)
            elif item_type == "TABLE":
                if current_content:
                    chunks.append(self._create_chunk(current_content, current_meta))
                    current_content = []
                
                table_id = self._extract_table_id(item.text)
                current_content.append(item.export_to_markdown())
                meta = {
                    "clause_id": current_meta["clause_id"] if current_meta else "unknown",
                    "title": item.caption.text if hasattr(item, 'caption') and item.caption else "Untitled Table",
                    "page": item.page_no,
                    "type": ContentType.TABLE,
                    "table_id": table_id
                }
                chunks.append(self._create_chunk(current_content, meta))
                current_content = []
            
            # 3. Handle Text/Formulas (Append to Current Concept)
            elif item_type in ["PARAGRAPH", "FORMULA", "LIST_ITEM"]:
                if item.text.strip():
                    current_content.append(item.text)
        
        if current_content:
            chunks.append(self._create_chunk(current_content, current_meta))
        
        return chunks

    def _create_chunk(self, content_lines: List[str], meta: Dict) -> DesignChunk:
        text = "\n".join(content_lines)
        chunk_id = f"{meta['clause_id']}_{uuid.uuid4().hex[:8]}"
        
        # Extract Cross-References for Graph
        references = []
        refs = self.clause_ref_pattern.findall(text)
        refs += self.table_pattern.findall(text)
        references = [f"clause_{r}" if not r.startswith("Table") else f"table_{r}" for r in refs]
        
        metadata = ChunkMetadata(
            chunk_id=chunk_id,
            content_type=meta['type'],
            canonical_source=CanonicalSource(
                clause_id=meta['clause_id'],
                clause_title=meta['title'],
                page_number=meta['page']
            ),
            references=references,
            table_caption=meta.get('table_id')
        )
        
        return DesignChunk(id=chunk_id, text=text, metadata=metadata)

    def _extract_clause_id(self, text: str) -> str:
        match = self.clause_pattern.match(text)
        return match.group(1) if match else "unknown"

    def _extract_table_id(self, text: str) -> str:
        match = self.table_pattern.search(text)
        return match.group(1) if match else "unknown"

from models import DesignChunk, RetrievalResult

class EngineeringRetriever:
    def __init__(self, indexer: EngineeringIndexer):
        self.indexer = indexer

    def retrieve(self, query: str, k: int = 5) -> List[RetrievalResult]:
        # 1. Query Routing (Simplified)
        # In production, use an LLM to classify query type
        is_design_query = any(term in query.lower() for term in ["design", "calculate", "how to"])
        
        # 2. Hierarchical Search
        # Step A: Find Relevant Sections (Filter Rule Index by Clause Prefix)
        # For this example, we search Rule Index directly with metadata filtering
        primary_chunks = self.indexer.rule_store.search(
            query=query, 
            filter={},  # Add clause filters here if sections identified
            k=k
        )
        
        # 3. Graph Expansion (CRITICAL FOR CITATIONS)
        expanded_results = []
        seen_ids = set()
        
        for chunk in primary_chunks:
            # Add Primary Result
            expanded_results.append(RetrievalResult(
                chunk=chunk,
                role="primary",
                relevance_score=1.0
            ))
            seen_ids.add(chunk.id)
            
            # Fetch References Dynamically
            if is_design_query:
                ref_ids = self.indexer.get_references(chunk.id)
                for ref_id in ref_ids:
                    if ref_id not in seen_ids:
                        ref_chunk = self.indexer.rule_store.get_by_id(ref_id)
                        if ref_chunk:
                            # CRITICAL: Keep ref_chunk's original metadata (canonical_source)
                            expanded_results.append(RetrievalResult(
                                chunk=ref_chunk,
                                role="reference",
                                relevance_score=0.8
                            ))
                            seen_ids.add(ref_id)
        
        return expanded_results

5. generator.py (Citation-Aware Context Assembly)
Formats the context for the LLM with explicit source tags.

from models import RetrievalResult
from typing import List

class ContextAssembler:
    def assemble(self, results: List[RetrievalResult], query: str) -> str:
        context = f"User Query: {query}\n\n"
        context += "## Retrieved Engineering Knowledge\n\n"
        
        # Group by Canonical Source for organized output
        by_source = {}
        for result in results:
            source = result.chunk.metadata.canonical_source
            key = f"{source.clause_id}"
            
            if key not in by_source:
                by_source[key] = {
                    "title": source.clause_title,
                    "page": source.page_number,
                    "items": []
                }
            
            by_source[key]["items"].append(result)
        
        # Format Context
        for clause_id, group in by_source.items():
            context += f"### [{clause_id}] {group['title']} (Page {group['page']})\n"
            
            for item in group["items"]:
                role_tag = "[PRIMARY]" if item.role == "primary" else "[REFERENCED]"
                context += f"{role_tag}:\n{item.chunk.text}\n\n"
        
        # Add Citation Instructions
        context += self._get_citation_instructions()
        return context

    def _get_citation_instructions(self) -> str:
        return """
## Citation Rules (STRICT)
1. **Primary Source:** If information comes from a [PRIMARY] block, cite: [Clause X.Y.Z, p. XX]
2. **Referenced Source:** If information comes from a [REFERENCED] block, cite the **Source Clause** listed in the header, NOT the clause that referenced it.
   - Example: If Table 6.1 is in Clause 6.1.1 but referenced by 6.1.2, cite [Table 6.1, Clause 6.1.1, p. 37].
3. **Equations:** Include Equation ID if available.
4. **Verification:** All claims must be traceable to the provided Clause IDs and Page Numbers.
"""

🛡️ Critical Implementation Notes for the Coding Agent
Metadata Immutability:
Rule: In retriever.py, when fetching referenced chunks, NEVER modify result.chunk.metadata.canonical_source.
Why: This ensures that if Table 6.1 (located in 6.1.1) is fetched because Clause 6.1.2 referenced it, the citation remains 6.1.1.
Graph Edge Storage:
Store references as Chunk IDs, not just text strings.
Why: Text matching ("Clause 6.1.2") is ambiguous. ID matching (chunk_6.1.2_v1) is precise.
Vector Store Filtering:
Ensure your vector store supports metadata filtering (e.g., filter={"clause_id": {"$startswith": "6.1"}}).
Why: This enables the hierarchical search step (finding all rules within Section 6).
Prompt Engineering:
The _get_citation_instructions in generator.py is as important as the retrieval logic.
Why: LLMs tend to cite the context they are reading now (the primary chunk) rather than the source of the data. Explicit instructions counteract this bias.
Docling Version:
Ensure docling is updated (pip install -U docling). The iterate_items() API may change between versions.
